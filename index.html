const recetas = {
normal: ["Pasta con tomate", "Pollo al horno", "Arroz con verduras"],
vegetariana: ["Ensalada de quinoa", "Tofu salteado", "Crema de lentejas"],
proteina: ["Pechuga de pollo", "Salmón a la plancha", "Omelette de claras"],
"baja-carb": ["Ensalada de pollo", "Carne asada", "Verduras al wok"]
};

function generarPlan() {
const tipo = document.getElementById("tipo-comida").value;
const plan = recetas[tipo];
let resultado = "<h3>Tu plan de comidas:</h3><ul>";
plan.forEach(plato => {
resultado += `<li>${plato}</li>`;
});
resultado += "</ul>";
document.getElementById("resultado-plan").innerHTML = resultado;
}
